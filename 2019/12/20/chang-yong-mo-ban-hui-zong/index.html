二分排序

int A(int l,int r)
{
    int mid=a[(l+r)/2];
    int i=l,j=r;
    while(i<=j)
    {
        while(a[i]mid&&j>=l)
            j--;
        if(i<=j)
        {
            swap(a[i],a[j]);
            i++;
            j--;
        }
    }
        if(l<=j)
            A(l,j);
        if(i<=r)
            A(i,r);
    return 0;
}
